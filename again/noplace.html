使用swiper插件完美实现图片的轮播效果：

                    var swiper = new Swiper('.swiper-container', {
                        grabCursor: true,
                        slidesPerView: 2,
                        touchRatio: 0.5,
                        loop: true,
                        speed: 1200,
                        autoplay: {
                            delay: 2000,
                            disableOnInteraction: false,
                        },
                        pagination: {
                            el: '.swiper-pagination',
                            clickable: true,
                            dynamicBullets: true,
                        },
                        navigation: {
                            nextEl: '.swiper-button-next',
                            prevEl: '.swiper-button-prev',
                        },
                    });
